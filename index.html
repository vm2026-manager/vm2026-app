function fillMissingSlotsOptimally() { const emptySlots = getEmptySlots(); if (!emptySlots.length) { showStatus("Der er ingen tomme pladser at udfylde.", "info"); return; } // På helt tomt hold bruges altid hurtig fallback. // Det forhindrer, at browseren fryser på GitHub Pages/mobil. if (emptySlots.length === getCurrentSlots().length) { applyFastFullSquadFallback("Tomt hold blev udfyldt med hurtig gyldig fallback-logik."); return; } const usedIds = new Set(Object.values(squad).filter(Boolean).map(String)); const teamCounts = getCurrentTeamCounts(); let remainingBank = BUDGET_TOTAL - getSpentBudget(); const positionRank = { "Angriber": 1, "Midtbane": 2, "Forsvar": 3, "Målmand": 4 }; const orderedSlots = [...emptySlots].sort((a, b) => { return (positionRank[a.position] || 9) - (positionRank[b.position] || 9); }); const additions = {}; for (let i = 0; i < orderedSlots.length; i++) { const slot = orderedSlots[i]; const remainingSlotsAfterThis = orderedSlots.slice(i + 1); let candidates = getEligiblePlayersForSlot(slot, usedIds, teamCounts, remainingBank); // Browser-sikker grænse. // Vi prøver de bedste kandidater, men undgår tung kombinationsoptimering i browseren. candidates = candidates.slice(0, 80); let chosen = null; for (const player of candidates) { const price = getPlayerPrice(player); const playerIdStr = String(player.player_id); const teamName = player.team_name || ""; const testUsedIds = new Set(usedIds); const testTeamCounts = { ...teamCounts }; testUsedIds.add(playerIdStr); testTeamCounts[teamName] = (testTeamCounts[teamName] || 0) + 1; const minimumRestCost = getCheapestRemainingCostForSlots( remainingSlotsAfterThis, testUsedIds, testTeamCounts ); if (price + minimumRestCost <= remainingBank) { chosen = player; break; } } // Hvis bedste score-valg ikke går op, prøv billigste gyldige kandidat. if (!chosen) { const cheapestCandidates = [...candidates].sort((a, b) => getPlayerPrice(a) - getPlayerPrice(b)); for (const player of cheapestCandidates) { const price = getPlayerPrice(player); const playerIdStr = String(player.player_id); const teamName = player.team_name || ""; const testUsedIds = new Set(usedIds); const testTeamCounts = { ...teamCounts }; testUsedIds.add(playerIdStr); testTeamCounts[teamName] = (testTeamCounts[teamName] || 0) + 1; const minimumRestCost = getCheapestRemainingCostForSlots( remainingSlotsAfterThis, testUsedIds, testTeamCounts ); if (price + minimumRestCost <= remainingBank) { chosen = player; break; } } } if (!chosen) { continue; } additions[slot.key] = chosen.player_id; usedIds.add(String(chosen.player_id)); teamCounts[chosen.team_name] = (teamCounts[chosen.team_name] || 0) + 1; remainingBank -= getPlayerPrice(chosen); } const addedCount = Object.keys(additions).length; if (!addedCount) { const blockingReasons = getBlockingReasonSummaryForSlots( emptySlots, new Set(Object.values(squad).filter(Boolean).map(String)), getCurrentTeamCounts(), BUDGET_TOTAL - getSpentBudget() ); showStatus( `Kunne ikke udfylde de tomme pladser under de nuværende constraints. ${blockingReasons.join(" · ")}`, "error" ); return; } const candidateSquad = { ...squad, ...additions }; const issues = getSquadConstraintIssues(candidateSquad); if (issues.length) { showStatus(`Udfyldningen blev stoppet, fordi resultatet ville være ugyldigt: ${issues.join(" · ")}`, "error"); return; } pushHistory(); squad = candidateSquad; activeSlotKey = null; activeSlotText.textContent = "Ingen valgt"; saveState(); render(); scheduleAutosave(); if (addedCount === emptySlots.length) { showStatus(`Udfyldte ${addedCount} tomme pladser med hurtig gyldig logik.`, "info"); } else { showStatus( `Udfyldte ${addedCount} af ${emptySlots.length} tomme pladser. De resterende pladser kunne ikke udfyldes uden at bryde budget, positioner eller maks. ${MAX_PER_TEAM} fra samme land.`, "info" ); } }
